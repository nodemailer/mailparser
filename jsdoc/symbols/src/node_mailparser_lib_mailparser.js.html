 19 // MailParser is a FSM - it is always in one of the possible states
 20 var STATES = {
 21     header:   0x1,
 22     body:     0x2,
 23     finished: 0x3
 24 };

 26 /**
 27  * <p>Creates instance of MailParser which in turn extends Stream</p>
 28  * 
 29  * <p>Options object has the following properties:</p>
 30  * 
 31  * <ul>
 32  *   <li><b>debug</b> - if set to true print all incoming lines to console</li>
 33  *   <li><b>streamAttachments</b> - if set to true, stream attachments instead of including them</li>
 34  *   <li><b>unescapeSMTP</b> - if set to true replace double dots in the beginning of the file</li>
 35  *   <li><b>defaultCharset</b> - the default charset for text/plain, text/html content, if not set reverts to Latin-1
 36  * </ul>
 37  * 
 38  * @constructor
 39  * @param {Object} [options] Optional options object
 40  */
 41 function MailParser(options){
 42     
 43     // Make MailParser a Stream object
 44     Stream.call(this);
 45     this.writable = true;
 46     
 47     /** 
 48      * Options object
 49      * @public  */ this.options = options || {};
 50      
 51     /** 
 52      * Indicates current state the parser is in
 53      * @private */ this._state         = STATES.header;
 54     
 55     /** 
 56      * The remaining data from the previos chunk which is waiting to be processed
 57      * @private */ this._remainder     = "";
 58     
 59     /** 
 60      * The complete tree structure of the e-mail
 61      * @public  */ this.mimeTree      = this._createMimeNode();
 62     
 63     /** 
 64      * Current node of the multipart mime tree that is being processed
 65      * @private */ this._currentNode   = this.mimeTree;
 66      
 67     /** 
 68      * An object of already used attachment filenames
 69      * @private */ this._fileNames     = {};
 70      
 71     /** 
 72      * An array of multipart nodes
 73      * @private */ this._multipartTree = [];
 74      
 75      
 76     /** 
 77      * Cache for iconv converter objects
 78      * @private */ this._iconv         = {};
 79      
 80     /**
 81      * This is the final mail structure object that is returned to the client
 82      * @public  */ this.mailData      = {};
 83      
 84     /** 
 85      * Line counter for debugging
 86      * @private */ this._lineCounter   = 0;
 87      
 88     /** 
 89      * Did the last chunk end with \r
 90      * @private */ this._lineFeed      = false;
 91 }
 92 // inherit methods and properties of Stream
 93 utillib.inherits(MailParser, Stream);

 95 /**
 96  * <p>Writes a value to the MailParser stream<p>
 97  * 
 98  * @param {Buffer|String} chunk The data to be written to the MailParser stream
 99  * @param {String} [encoding] The encoding to be used when "chunk" is a string
100  * @returns {Boolean} Returns true
101  */
102 MailParser.prototype.write = function(chunk, encoding){
103     if(typeof chunk == "string"){
104         chunk = new Buffer(chunk, encoding);
105     }
106     
107     if(chunk && chunk.length){
108         this._remainder += chunk.toString("binary");
109         process.nextTick(this._process.bind(this));
110         return true;
111     }else{
112         return true;
113     }
114 };
115 
116 /**
117  * <p>Terminates the MailParser stream</p> 
118  * 
119  * <p>If "chunk" is set, writes it to the Stream before terminating.</p>
120  * 
121  * @param {Buffer|String} chunk The data to be written to the MailParser stream
122  * @param {String} [encoding] The encoding to be used when "chunk" is a string
123  */
124 MailParser.prototype.end = function(chunk, encoding){
125     if(typeof chunk == "string"){
126         chunk = new Buffer(chunk, encoding);
127     }
128     
129     if(this.options.debug && this._remainder){
130         console.log("REMAINDER: "+this._remainder);
131     }
132     
133     chunk = chunk && chunk.toString("binary") || "";
134     
135     // if the last chunk ended with \r and this one begins
136     // with \n, it's a split line ending. Since the last \r
137     // was already used, skip the \n
138     if(this._lineFeed && chunk.charAt(0) == "\n"){
139         chunk = chunk.substr(1);
140     }
141     this._lineFeed = chunk.substr(-1) == "\r";
142     
143     this._remainder += chunk;
144 
145     process.nextTick(this._process.bind(this, true));
146 };
147 
148 /**
149  * <p>Processes the data written to the MailParser stream</p>
150  * 
151  * <p>The data is split into lines and each line is processed individually. Last
152  * line in the batch is preserved as a remainder since it is probably not a
153  * complete line but just the beginning of it. The remainder is later prepended
154  * to the next batch of data.</p>
155  * 
156  * @param {Boolean} [finalPart=false] if set to true indicates that this is the last part of the stream
157  */
158 MailParser.prototype._process = function(finalPart){
159     
160     finalPart = !!finalPart;
161     
162     var lines = this._remainder.split(/\r?\n|\r/),
163         line, i, len;
164         
165     if(!finalPart){
166         this._remainder = lines.pop();
167         // force line to 1MB chunks if needed 
168         if(this._remainder.length>1048576){
169             this._remainder = this._remainder.replace(/(.{1048576}(?!\r?\n|\r))/g,"$&\n");
170         }
171     }
172     
173     for(i=0, len=lines.length; i < len; i++){
174         line = lines[i];
175         
176         if(this.options.unescapeSMTP && line.substr(0,2)==".."){
177             line = line.substr(1);
178         }
179         
180         if(this.options.debug){
181             console.log("LINE " + (++this._lineCounter) + " ("+this._state+"): "+line);
182         }
183         
184         if(this._state == STATES.header){
185             if(this._processStateHeader(line) === true){
186                 continue;
187             }
188         }
189         
190         if(this._state == STATES.body){
191             
192             if(this._processStateBody(line) === true){
193                 continue;
194             }
195             
196         }
197     }
198     
199     if(finalPart){
200         if(this._state == STATES.header && this._remainder){
201             this._processStateHeader(this._remainder);
202         }
203         if(this._currentNode.content || this._currentNode.stream){
204             this._finalizeContents();
205         }
206         this._state = STATES.finished;
207         process.nextTick(this._processMimeTree.bind(this));
208     }
209     
210     
211 };

213 /**
214  * <p>Processes a line while in header state</p>
215  * 
216  * <p>If header state ends and body starts, detect if the contents is an attachment
217  * and create a stream for it if needed</p>
218  * 
219  * @param {String} line The contents of a line to be processed
220  * @returns {Boolean} If state changes to body retuns true
221  */
222 MailParser.prototype._processStateHeader = function(line){
223     var boundary, i, len, attachment, 
224         lastPos = this._currentNode.headers.length - 1,
225         textContent = false;
226     
227     // Check if the header endas and body starts
228     if(!line.length){
229         this._state = STATES.body;
230         
231         // if there's unprocessed header data, do it now
232         if(lastPos >= 0){
233             this._processHeaderLine(lastPos);
234         }
235         
236         // this is a very simple e-mail, no content type set
237         if(!this._currentNode.parentNode && !this._currentNode.meta.contentType){
238             this._currentNode.meta.contentType = "text/plain";
239         }
240         
241         textContent = ["text/plain", "text/html"].indexOf(this._currentNode.meta.contentType || "") >= 0;
242         
243         // detect if this is an attachment or a text node (some agents use inline dispositions for text)
244         if(textContent && (!this._currentNode.meta.contentDisposition || this._currentNode.meta.contentDisposition == "inline")){
245             this._currentNode.attachment = false;
246         }else if((!textContent || ["attachment", "inline"].indexOf(this._currentNode.meta.contentDisposition)>=0) && 
247           !this._currentNode.meta.mimeMultipart){
248             this._currentNode.attachment = true;
249         }
250         
251         // handle attachment start
252         if(this._currentNode.attachment){
253             
254             this._currentNode.meta.length = 0;
255             this._currentNode.checksum = crypto.createHash("md5");
256             
257             fileName = this._currentNode.meta.fileName || "attachment";
258             this._currentNode.meta.generatedFileName = this._generateFileName(fileName);
259             
260             attachment = {
261                 fileName: fileName
262             };
263             
264             if(this.options.streamAttachments){
265                 if(this._currentNode.meta.contentType){
266                     attachment.contentType = this._currentNode.meta.contentType || null;
267                 }
268                 
269                 if(this._currentNode.meta.contentId){
270                     attachment.contentId = this._currentNode.meta.contentId || "";
271                 }
272                 
273                 if(this._currentNode.meta.charset){
274                     attachment.charset = this._currentNode.meta.charset;
275                 }
276             
277                 if(this._currentNode.meta.transferEncoding == "base64"){
278                     this._currentNode.stream = new Streams.Base64Stream();
279                 }else if(this._currentNode.meta.transferEncoding == "quoted-printable"){
280                     this._currentNode.stream = new Streams.QPStream("binary");
281                 }else{
282                     this._currentNode.stream = new Streams.BinaryStream();
283                 }
284                 attachment.stream = this._currentNode.stream;
285                 
286                 this.emit("attachment", attachment);
287             }else{
288                 this._currentNode.content = undefined;
289             }
290         }
291         
292         return true;
293     }
294     
295     // unfold header lines if needed
296     if(line.match(/^\s+/) && lastPos>=0){
297         this._currentNode.headers[lastPos] += " " + line.trim();
298     }else{
299         this._currentNode.headers.push(line.trim());
300         if(lastPos>=0){
301             // if a complete header line is received, process it
302             this._processHeaderLine(lastPos);
303         }
304     }
305     
306     return false;
307 };

309 /**
310  * <p>Processes a line while in body state</p>
311  * 
312  * @param {String} line The contents of a line to be processed
313  * @returns {Boolean} If body ends return true
314  */
315 MailParser.prototype._processStateBody = function(line){
316     var i, len, node,
317         nodeReady = false;
318     
319     // Handle multipart boundaries
320     if(line.substr(0, 2) == "--"){
321         for(i=0, len = this._multipartTree.length; i<len; i++){
322             
323             // check if a new element block starts
324             if(line == "--" + this._multipartTree[i].boundary){
325                 
326                 if(this._currentNode.content || this._currentNode.stream){
327                     this._finalizeContents();
328                 }
329                 
330                 node = this._createMimeNode(this._multipartTree[i].node);
331                 this._multipartTree[i].node.childNodes.push(node);
332                 this._currentNode = node;
333                 this._state = STATES.header;
334                 nodeReady = true;
335                 break;
336             }else 
337             // check if a multipart block ends
338               if(line == "--" + this._multipartTree[i].boundary + "--"){
339                 
340                 if(this._currentNode.content || this._currentNode.stream){
341                     this._finalizeContents();
342                 }
343                 
344                 if(this._multipartTree[i].node.parentNode){
345                     this._currentNode = this._multipartTree[i].node.parentNode;
346                 }else{
347                     this._currentNode = this._multipartTree[i].node;
348                 }
349                 this._state = STATES.body;
350                 nodeReady = true;
351                 break;
352             }
353         }
354     }
355     if(nodeReady){
356         return true;
357     }
358     
359     // handle text or attachment line
360     if(["text/plain", "text/html"].indexOf(this._currentNode.meta.contentType || "")>=0 && 
361       !this._currentNode.attachment){
362         this._handleTextLine(line);
363     }else if(this._currentNode.attachment){
364         this._handleAttachmentLine(line);
365     }
366     
367     return false;
368 };

370 /**
371  * <p>Processes a complete unfolded header line</p>
372  * 
373  * <p>Processes a line from current node headers array and replaces its value.
374  * Input string is in the form of "X-Mailer: PHP" and its replacement would be
375  * an object <code>{key: "x-mailer", value: "PHP"}</code></p>
376  * 
377  * <p>Additionally node meta object will be filled also, for example with data from
378  * To: From: Cc: etc fields.</p>
379  * 
380  * @param {Number} pos Which header element (from an header lines array) should be processed
381  */
382 MailParser.prototype._processHeaderLine = function(pos){
383     var key, value, parts, line;
384     
385     pos = pos || 0;
386     
387     if(!(line = this._currentNode.headers[pos])){
388         return;
389     }
390     
391     parts = line.split(":");
392     
393     key = parts.shift().toLowerCase().trim();
394     value = parts.join(":").trim();
395     
396     switch(key){
397         case "content-type":
398             this._parseContentType(value);
399             break;
400         case "mime-version":
401             this._currentNode.useMIME = true;
402             break;
403         case "date":
404             this._currentNode.meta.date = new Date(datetime.strtotime(value)*1000 || Date.now());
405             break;
406         case "to":
407             if(this._currentNode.to && this._currentNode.to.length){
408                 this._currentNode.to = this._currentNode.to.concat(mimelib.parseAddresses(value));
409             }else{
410                 this._currentNode.to = mimelib.parseAddresses(value);
411             }
412             break;
413         case "from":
414             if(this._currentNode.from && this._currentNode.from.length){
415                 this._currentNode.from = this._currentNode.from.concat(mimelib.parseAddresses(value));
416             }else{
417                 this._currentNode.from = mimelib.parseAddresses(value);
418             }
419             break;
420         case "cc":
421             if(this._currentNode.cc && this._currentNode.cc.length){
422                 this._currentNode.cc = this._currentNode.cc.concat(mimelib.parseAddresses(value));
423             }else{
424                 this._currentNode.cc = mimelib.parseAddresses(value);
425             }
426             break;
427         case "bcc":
428             if(this._currentNode.bcc && this._currentNode.bcc.length){
429                 this._currentNode.bcc = this._currentNode.bcc.concat(mimelib.parseAddresses(value));
430             }else{
431                 this._currentNode.bcc = mimelib.parseAddresses(value);
432             }
433             break;
434         case "x-priority":
435         case "x-msmail-priority":
436         case "importance":
437             value = this._parsePriority(value);
438             break;
439         case "message-id":
440             this._currentNode.meta.messageId = this._trimQuotes(value);
441             break;
442         case "references":
443             this._currentNode.meta.messageReferences = this._trimQuotes(value);
444             break;
445         case "in-reply-to":
446             this._currentNode.meta.inReplyTo = this._trimQuotes(value);
447             break;
448         case "thread-index":
449             this._currentNode.meta.threadIndex = value;
450             break;
451         case "content-transfer-encoding":
452             this._currentNode.meta.transferEncoding = value.toLowerCase();
453             break;
454         case "subject":
455             this._currentNode.subject = this._encodeString(value);
456             break;
457         case "content-disposition":
458             this._parseContentDisposition(value);
459             break;
460         case "content-id":
461             this._currentNode.meta.contentId = this._trimQuotes(value);
462             break;
463     }
464     
465     if(this._currentNode.parsedHeaders[key]){
466         if(!Array.isArray(this._currentNode.parsedHeaders[key])){
467             this._currentNode.parsedHeaders[key] = [this._currentNode.parsedHeaders[key]];
468         } 
469         this._currentNode.parsedHeaders[key].push(value);
470     }else{
471         this._currentNode.parsedHeaders[key] = value;
472     }
473     
474     this._currentNode.headers[pos] = {key: key, value: value};
475 };

477 /**
478  * <p>Creates an empty node element for the mime tree</p>
479  * 
480  * <p>Created element includes parentNode property and a childNodes array. This is
481  * needed to later walk the whole mime tree</p>
482  * 
483  * @param {Object} [parentNode] the parent object for the created node
484  * @returns {Object} node element for the mime tree
485  */
486 MailParser.prototype._createMimeNode = function(parentNode){
487     var node = {
488         parentNode: parentNode || this._currentNode || null,
489         headers: [],
490         parsedHeaders:{},
491         meta: {},
492         childNodes: []
493     };
494     
495     return node;
496 };

498 /**
499  * <p>Splits a header value into key-value pairs</p>
500  * 
501  * <p>Splits on <code>;</code> - the first value will be set as <code>defaultValue</code> property and will
502  * not be handled, others will be split on <code>=</code> to key-value pairs</p>
503  * 
504  * <p>For example <code>content-type: text/plain; charset=utf-8</code> will become:</p>
505  * 
506  * <pre>
507  * {
508  *     defaultValue: "text/plain",
509  *     charset: "utf-8"
510  * }
511  * </pre>
512  * 
513  * @param {String} value A string to be splitted into key-value pairs
514  * @returns {Object} a key-value object, with defaultvalue property
515  */
516 MailParser.prototype._parseHeaderLineWithParams = function(value){
517     var key, parts, returnValue = {};
518     
519     parts = value.split(";");
520     returnValue.defaultValue = parts.shift().toLowerCase();
521     
522     for(var i=0, len = parts.length; i<len; i++){
523         value = parts[i].split("=");
524         key = value.shift().trim().toLowerCase();
525         value = value.join("=").trim();
526         
527         // trim quotes
528         value = this._trimQuotes(value);
529         returnValue[key] = value;
530     }
531     
532     return returnValue;
533 };

535 /**
536  * <p>Parses a Content-Type header field value</p>
537  * 
538  * <p>Fetches additional properties from the content type (charset etc.) and fills
539  * current node meta object with this data</p>
540  * 
541  * @param {String} value Content-Type string
542  * @returns {Object} parsed contenttype object
543  */
544 MailParser.prototype._parseContentType = function(value){
545     value = this._parseHeaderLineWithParams(value);
546     if(value){
547         if(value.defaultValue){
548             value.defaultValue = value.defaultValue.toLowerCase();
549             this._currentNode.meta.contentType = value.defaultValue;
550             if(value.defaultValue.substr(0,"multipart/".length)=="multipart/"){
551                 this._currentNode.meta.mimeMultipart = value.defaultValue.substr("multipart/".length);
552             }
553         }else{
554             this._currentNode.meta.contentType = "application/octet-stream";
555         }
556         if(value.charset){
557             value.charset = value.charset.toLowerCase();
558             if(value.charset.substr(0,4)=="win-"){
559                 value.charset = "windows-"+value.charset.substr(4); 
560             }else if(value.charset.match(/^utf\d/)){
561                 value.charset = "utf-"+value.charset.substr(3);
562             }else if(value.charset.match(/^latin[\-_]?\d/)){
563                 value.charset = "iso-8859-"+value.charset.replace(/\D/g,""); 
564             }else if(value.charset.match(/^(us\-)?ascii$/)){
565                 value.charset = "utf-8"; 
566             }  
567             this._currentNode.meta.charset = value.charset;
568         }
569         if(value.format){
570             this._currentNode.meta.textFormat = value.format;
571         }
572         if(value.boundary){
573             this._currentNode.meta.mimeBoundary = value.boundary;
574         }
575         if(value.name && !this._currentNode.meta.fileName){
576             this._currentNode.meta.fileName = this._replaceMimeWords(value.name);
577         }
578         if(value.boundary){
579             this._currentNode.meta.mimeBoundary = value.boundary;
580             this._multipartTree.push({
581                 boundary: value.boundary,
582                 node: this._currentNode
583             });
584         }
585     } 
586     return value;
587 };

589 /**
590  * <p>Parses Content-Disposition header field value</p>
591  * 
592  * <p>Fetches filename to current node meta object</p>
593  * 
594  * @param {String} value A Content-Disposition header field
595  */
596 MailParser.prototype._parseContentDisposition = function(value){
597     var returnValue = {};
598     
599     value = this._parseHeaderLineWithParams(value);
600     
601     if(value){
602         if(value.defaultValue){
603             this._currentNode.meta.contentDisposition = value.defaultValue.trim().toLowerCase();
604         }
605         if(value.filename){
606             this._currentNode.meta.fileName = this._replaceMimeWords(value.filename);
607         }
608     }
609 };

611 /**
612  * <p>Parses the priority of the e-mail</p>
613  * 
614  * @param {String} value The priority value
615  * @returns {String} priority string low|normal|high
616  */
617 MailParser.prototype._parsePriority = function(value){
618     value = value.toLowerCase().trim();
619     if(!isNaN(value)){
620         value = Number(value) || 0;
621         if(value == 3){
622             return "normal";
623         }else if(value > 3){
624             return "low";
625         }else{
626             return "high";
627         }
628     }else{
629         switch(value){
630             case "non-urgent":
631             case "low":
632                 return "low";
633             case "urgent":
634             case "hight":
635                 return "high";
636         }
637     }
638     return "normal";
639 };

641 /**
642  * <p>Processes a line in text/html or text/plain node</p>
643  * 
644  * <p>Append the line to the content property</p>
645  * 
646  * @param {String} line A line to be processed 
647  */
648 MailParser.prototype._handleTextLine = function(line){
649     
650     if(["quoted-printable", "base64"].indexOf(this._currentNode.meta.transferEncoding)>=0 || this._currentNode.meta.textFormat != "flowed"){
651         if(typeof this._currentNode.content != "string"){
652             this._currentNode.content = line;
653         }else{
654             this._currentNode.content += "\n"+line;
655         }
656     }else{
657         if(typeof this._currentNode.content != "string"){
658             this._currentNode.content = line;
659         }else if(this._currentNode.content.match(/[ ]{1,}$/)){
660             this._currentNode.content += line;
661         }else{
662             this._currentNode.content += "\n"+line;
663         }
664     }  
665 };

667 /**
668  * <p>Processes a line in an attachment node</p>
669  * 
670  * <p>If a stream is set up for the attachment write the line to the
671  * stream as a Buffer object, otherwise append it to the content property</p>
672  * 
673  * @param {String} line A line to be processed 
674  */
675 MailParser.prototype._handleAttachmentLine = function(line){
676     if(!this._currentNode.attachment){
677         return;
678     }
679     if(this._currentNode.stream){
680         this._currentNode.stream.write(new Buffer(line, "binary"));
681     }else if("content" in this._currentNode){
682         if(typeof this._currentNode.content!="string"){
683             this._currentNode.content = line;
684         }else{
685             this._currentNode.content += "\r\n" + line;
686         }
687     }
688 };

690 /**
691  * <p>Finalizes a node processing</p>
692  * 
693  * <p>If the node is a text/plain or text/html, convert it to UTF-8 encoded string
694  * If it is an attachment, convert it to a Buffer or if an attachment stream is
695  * set up, close the stream</p>
696  */
697 MailParser.prototype._finalizeContents = function(){
698     var streamInfo;
699 
700     if(this._currentNode.content){
701         
702         if(!this._currentNode.attachment){
703             
704             if(this._currentNode.meta.contentType == "text/html"){
705                  this._currentNode.meta.charset = this._detectHTMLCharset(this._currentNode.content) || this._currentNode.meta.charset || this.options.defaultCharset || "iso-8859-1";
706             }
707             
708             if(this._currentNode.meta.transferEncoding == "quoted-printable"){
709                 this._currentNode.content = mimelib.decodeQuotedPrintable(this._currentNode.content, false, this._currentNode.meta.charset || this.options.defaultCharset || "iso-8859-1");
710             }else if(this._currentNode.meta.transferEncoding == "base64"){
711                 this._currentNode.content = mimelib.decodeBase64(this._currentNode.content, false, this._currentNode.meta.charset || this.options.defaultCharset || "iso-8859-1");
712             }else{
713                 this._currentNode.content = this._encodeString(this._currentNode.content);
714             }
715         }else{
716             if(this._currentNode.meta.transferEncoding == "quoted-printable"){
717                 this._currentNode.content = mimelib.decodeQuotedPrintable(this._currentNode.content, false, "binary");
718             }else if(this._currentNode.meta.transferEncoding == "base64"){
719                 this._currentNode.content = new Buffer(this._currentNode.content.replace(/[^\w\+\/=]/g,''), "base64");
720             }else{
721                 this._currentNode.content = new Buffer(this._currentNode.content, "binary");
722             }
723             this._currentNode.checksum.update(this._currentNode.content);
724             this._currentNode.meta.checksum = this._currentNode.checksum.digest("hex");
725             this._currentNode.meta.length = this._currentNode.content.length;
726         }
727         
728     }
729 
730     if(this._currentNode.stream){
731         streamInfo = this._currentNode.stream.end() || {};
732         if(streamInfo.checksum){
733             this._currentNode.meta.checksum = streamInfo.checksum;
734         }
735         if(streamInfo.length){
736             this._currentNode.meta.length = streamInfo.length;
737         }
738     }
739 };

741 /**
742  * <p>Processes the mime tree</p>
743  * 
744  * <p>Finds text parts and attachments from the tree. If there's several text/plain
745  * or text/html parts, push the ones from the lower parts of the tree to the
746  * alternatives array</p>
747  * 
748  * <p>Emits "end" when finished</p>
749  */
750 MailParser.prototype._processMimeTree = function(){
751     var level = 0, htmlLevel, textLevel, html, text,
752         returnValue = {}, i, len;
753     
754     this.mailData = {html:[], text:[], alternatives:[], attachments:[]};
755     
756     if(!this.mimeTree.meta.mimeMultipart){
757         this._processMimeNode(this.mimeTree, 0);
758     }else{
759         this._walkMimeTree(this.mimeTree);
760     }
761     
762     if(this.mailData.html.length){
763         for(i=0, len=this.mailData.html.length; i<len; i++){
764             if(!returnValue.html || this.mailData.html[i].level < htmlLevel){
765                 if(html){
766                     if(!returnValue.alternatives){
767                         returnValue.alternatives = [];
768                     }
769                     returnValue.alternatives.push({
770                         contentType: "text/html",
771                         content: html
772                     });
773                 }
774                 htmlLevel = this.mailData.html[i].level;
775                 html = this.mailData.html[i].content;
776             }
777         }
778     }
779     
780     if(this.mailData.text.length){
781         for(i=0, len=this.mailData.text.length; i<len; i++){
782             if(!returnValue.text || this.mailData.text[i].level < textLevel){
783                 if(text){
784                     if(!returnValue.alternatives){
785                         returnValue.alternatives = [];
786                     }
787                     returnValue.alternatives.push({
788                         contentType: "text/plain",
789                         content: text
790                     });
791                 }
792                 textLevel = this.mailData.text[i].level;
793                 text = this.mailData.text[i].content;
794             }
795         }
796     }
797     
798     returnValue.headers = this.mimeTree.parsedHeaders;
799     
800     if(this.mimeTree.subject){
801         returnValue.subject = this.mimeTree.subject;        
802     }
803     
804     if(this.mimeTree.from){
805         returnValue.from = this.mimeTree.from;        
806     }
807     
808     if(this.mimeTree.to){
809         returnValue.to = this.mimeTree.to;        
810     }
811     
812     if(this.mimeTree.cc){
813         returnValue.cc = this.mimeTree.cc;        
814     }
815     
816     if(this.mimeTree.bcc){
817         returnValue.bcc = this.mimeTree.bcc;        
818     }
819     
820     if(html){
821         returnValue.html = html;
822     }
823     
824     if(text){
825         returnValue.text = text;
826     }
827     
828     if(this.mailData.attachments.length){
829         returnValue.attachments = [];
830         for(i=0, len=this.mailData.attachments.length; i<len; i++){
831             returnValue.attachments.push(this.mailData.attachments[i].content);
832         }
833     }
834 
835     process.nextTick(this.emit.bind(this, "end", returnValue));
836 };

838 /**
839  * <p>Walks the mime tree and runs processMimeNode on each node of the tree</p>
840  * 
841  * @param {Object} node A mime tree node
842  * @param {Number} [level=0] current depth
843  */
844 MailParser.prototype._walkMimeTree = function(node, level){
845     level = level || 1;
846 
847     for(var i=0, len = node.childNodes.length; i<len; i++){
848         this._processMimeNode(node.childNodes[i], level);
849         this._walkMimeTree(node.childNodes[i], level+1);
850     }
851 };
852 
853 /**
854  * <p>Processes of a node in the mime tree</p>
855  * 
856  * <p>Pushes the node into appropriate <code>this.mailData</code> array (<code>text/html</code> to <code>this.mailData.html</code> array etc)</p>
857  * 
858  * @param {Object} node A mime tree node
859  * @param {Number} [level=0] current depth
860  */
861 MailParser.prototype._processMimeNode = function(node, level){
862     level = level || 0;
863 
864     if(!node.attachment){
865         switch(node.meta.contentType){
866             case "text/html":
867                 this.mailData.html.push({content: this._updateHTMLCharset(node.content || ""), level: level});
868                 break;
869             case "text/plain":
870                 this.mailData.text.push({content: node.content || "", level: level});
871                 break;
872         }
873     }else{
874         node.meta = node.meta || {};
875         if(node.content){
876             node.meta.content = node.content;
877         }
878         this.mailData.attachments.push({content: node.meta || {}, level: level});
879     }  
880 };

882 /**
883  * <p>Converts a string from one charset to another</p>
884  * 
885  * @param {Buffer|String} value A String to be converted
886  * @param {String} fromCharset source charset
887  * @param {String} [toCharset="UTF-8"] destination charset
888  * @returns {Buffer} Converted string as a Buffer (or SlowBuffer)
889  */
890 MailParser.prototype._convertString = function(value, fromCharset, toCharset){
891     toCharset = (toCharset || "utf-8").toUpperCase();
892     fromCharset = (fromCharset || "utf-8").toUpperCase();
893     
894     value = typeof value=="string"?new Buffer(value, "binary"):value;
895     
896     if(toCharset == fromCharset){
897         return value;
898     }
899     
900     try{ // in case there is no such charset or EINVAL occurs leave the string untouched
901         if(!this._iconv[fromCharset+toCharset]){
902             this._iconv[fromCharset+toCharset] = new Iconv(fromCharset, toCharset+'//TRANSLIT//IGNORE');
903         }
904         value = this._iconv[fromCharset+toCharset].convert(value);
905     }catch(E){}
906     
907     return value;
908 };

910 /**
911  * <p>Encodes a header string to UTF-8</p>
912  * 
913  * @param {String} value String to be encoded
914  * @returns {String} UTF-8 encoded string
915  */
916 MailParser.prototype._encodeString = function(value){
917     value = this._replaceMimeWords(this._convertString(value, this._currentNode.meta.charset || this.options.defaultCharset || "iso-8859-1").toString("utf-8"));
918     return value;
919 };

921 /**
922  * <p>Replaces mime words in a string with UTF-8 encoded strings</p>
923  * 
924  * @param {String} value String to be converted
925  * @returns {String} converted string
926  */
927 MailParser.prototype._replaceMimeWords = function(value){
928     return value.
929         replace(/(=\?[^?]+\?[QqBb]\?[^?]+\?=)\s+(?==\?[^?]+\?[QqBb]\?[^?]+\?=)/g, "$1"). // join mimeWords
930         replace(/\=\?[^?]+\?[QqBb]\?[^?]+\?=/g, (function(a){
931             return mimelib.decodeMimeWord(a);
932         }).bind(this));
933 };

935 /**
936  * <p>Removes enclosing quotes ("", '', <>) from a string</p>
937  * 
938  * @param {String} value String to be converted
939  * @returns {String} converted string
940  */
941 MailParser.prototype._trimQuotes = function(value){
942     value = (value || "").trim();
943     if((value.charAt(0)=='"' && value.charAt(value.length-1)=='"') || 
944       (value.charAt(0)=="'" && value.charAt(value.length-1)=="'") || 
945       (value.charAt(0)=="<" && value.charAt(value.length-1)==">")){
946         value = value.substr(1,value.length-2);
947     }
948     return value;
949 };

951 /**
952  * <p>Generates a context unique filename for an attachment</p>
953  * 
954  * <p>If a filename already exists, append a number to it</p>
955  * 
956  * <ul>
957  *     <li>file.txt</li>
958  *     <li>file-1.txt</li>
959  *     <li>file-2.txt</li>
960  * </ul>
961  * 
962  * @param {String} fileName source filename
963  * @returns {String} generated filename
964  */
965 MailParser.prototype._generateFileName = function(fileName){
966     var ext;
967     
968     // remove path if it is included in the filename
969     fileName = (fileName || "attachment").toString().split(/[\/\\]+/).pop().replace(/^\.+/,"") || "attachment";
970     
971     if(fileName in this._fileNames){
972         this._fileNames[fileName]++;
973         ext = fileName.substr((fileName.lastIndexOf(".") || 0)+1);
974         if(ext == fileName){
975             fileName += "-" +  this._fileNames[fileName];
976         }else{
977             fileName = fileName.substr(0, fileName.length - ext.length - 1) + "-" + this._fileNames[fileName] + "." + ext;
978         }
979     }else{
980         this._fileNames[fileName] = 0;
981     }
982     return fileName;  
983 };

986 /**
987  * <p>Replaces character set to UTF-8 in HTML <meta> tags</p>
988  * 
989  * @param {String} HTML html contents
990  * @returns {String} updated HTML
991  */
992 MailParser.prototype._updateHTMLCharset = function(html){
993     
994     html = html.replace(/\n/g,"\u0000").
995         replace(/<meta[^>]*>/gi, function(meta){
996             if(meta.match(/http\-equiv\s*=\s*"?content\-type/i)){
997                 return '<meta http-equiv="content-type" content="text/html; charset=utf-8" />';
998             }
999             if(meta.match(/\scharset\s*=\s*['"]?[\w\-]+["'\s>\/]/i)){
1000                 return '<meta charset="utf-8"/>';
1001             }
1002             return meta;
1003         }).
1004         replace(/\u0000/g,"\n");
1005     
1006     return html;
1007 };

1009 /**
1010  * <p>Detects the charset of an HTML file</p>
1011  * 
1012  * @param {String} HTML html contents
1013  * @returns {String} Charset for the HTML
1014  */
1015 MailParser.prototype._detectHTMLCharset = function(html){
1016     var charset, input, meta;
1017     
1018     if(typeof html !=" string"){
1019         html = html.toString("ascii");
1020     }
1021     
1022     if((meta = html.match(/<meta\s+http-equiv=["']content-type["'][^>]*?>/i))){
1023         input = meta[0];
1024     }
1025 
1026     if(input){
1027         charset = input.match(/charset\s?=\s?([a-zA-Z\-_:0-9]*);?/);
1028         if(charset){
1029             charset = (charset[1] || "").trim().toLowerCase();
1030         }
1031     }
1032 
1033     if(!charset && (meta = html.match(/<meta\s+charset=["']([^'"<\/]*?)["']/i))){
1034         charset = (meta[1] || "").trim().toLowerCase();
1035     }
1036 
1037     return charset;
1038 };
